# 创建存储过程

DROP PROCEDURE IF EXISTS get_customer_by_id;

DELIMITER //

CREATE PROCEDURE get_customer_by_id (
    id VARCHAR(10)
)
BEGIN
    SELECT * FROM  customers c
    WHERE c.cust_id = id;
END//

DELIMITER ;

再调用：

CALL get_customer_by_id(10003)
